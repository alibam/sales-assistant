## 任务 1：升级 Prisma 数据库支持企业私有化 RAG

### 1.1 启用 PostgreSQL Vector 扩展

在 `prisma/schema.prisma` 中：
```prisma
generator client {
  provider        = "prisma-client-js"
  previewFeatures = ["postgresqlExtensions"]
}

datasource db {
  provider   = "postgresql"
  url        = env("DATABASE_URL")
  extensions = [vector]
}
```

### 1.2 新增 KnowledgeBase 模型

```prisma
model KnowledgeBase {
  id          String   @id @default(uuid()) @db.Uuid
  tenantId    String   @db.Uuid  // 数据隔离
  name        String
  description String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  
  // 关联
  chunks      DocumentChunk[]
  
  @@index([tenantId])
  @@map("knowledge_bases")
}
```

### 1.3 新增 DocumentChunk 模型

```prisma
model DocumentChunk {
  id              String   @id @default(uuid()) @db.Uuid
  tenantId        String   @db.Uuid  // 数据隔离
  knowledgeBaseId String   @db.Uuid
  content         String   @db.Text
  metadata        Json?    // 存储文档来源、页码等元数据
  
  // Embedding 向量（使用 Qwen3-embedding）
  // Qwen3-embedding 输出维度通常是 1024 或 1536
  // 使用 Unsupported 类型以支持 pgvector
  embedding       Unsupported("vector(1024)")?
  
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
  
  // 关联
  knowledgeBase   KnowledgeBase @relation(fields: [knowledgeBaseId], references: [id], onDelete: Cascade)
  
  @@index([tenantId])
  @@index([knowledgeBaseId])
  @@map("document_chunks")
}
```

### 1.4 更新 .env.example

在 `.env.example` 中添加：
```env
# RAG Embedding API (私有化部署的 Qwen3-embedding)
EMBEDDING_API_URL=http://localhost:8000/v1/embeddings
EMBEDDING_API_KEY=your-embedding-api-key-here
EMBEDDING_MODEL=qwen3-embedding
```

### 1.5 生成 Prisma Client

```bash
npx prisma generate
npx prisma migrate dev --name add_rag_knowledge_base
```

## 任务 2：创建多角色 Super Dashboard 骨架

### 2.1 修改 app/(dashboard)/page.tsx

```typescript
import { requireAuth } from '@/lib/auth/session';
import SalesRepDashboard from './components/SalesRepDashboard';
import ManagerDashboard from './components/ManagerDashboard';
import AdminDashboard from './components/AdminDashboard';

export default async function DashboardPage() {
  const session = await requireAuth();
  
  // 根据角色分流到不同的看板
  switch (session.role) {
    case 'SALES_REP':
      return <SalesRepDashboard session={session} />;
    case 'SALES_MANAGER':
      return <ManagerDashboard session={session} />;
    case 'TENANT_ADMIN':
      return <AdminDashboard session={session} />;
    default:
      return <div>未知角色</div>;
  }
}
```

## 任务 3：编写 BDD 路由与展示测试

### 3.1 创建测试文件

文件：`tests/dashboard-role-routing.spec.ts`

```typescript
import { test, expect } from '@playwright/test';

test.describe('BDD: Dashboard 角色路由与权限分流', () => {
  test('Given 销售代表登录, When 访问首页, Then 显示销售代表看板', async ({ page }) => {
    // Given: 使用销售代表账号登录
    await page.goto('/api/auth/login?username=demo-user');
    
    // When: 访问首页
    await page.goto('/');
    
    // Then: 验证显示销售代表看板
    await expect(page.getByRole('heading', { name: '销售代表看板' })).toBeVisible();
    await expect(page.getByText('A 级客户')).toBeVisible();
    await expect(page.getByText('待跟进客户')).toBeVisible();
    
    // 验证不显示其他角色的内容
    await expect(page.getByText('销售经理看板')).not.toBeVisible();
    await expect(page.getByText('管理员看板')).not.toBeVisible();
  });
  
  test('Given 销售经理登录, When 访问首页, Then 显示销售经理看板', async ({ page }) => {
    // Given: 使用销售经理账号登录
    await page.goto('/api/auth/login?username=manager-user');
    
    // When: 访问首页
    await page.goto('/');
    
    // Then: 验证显示销售经理看板
    await expect(page.getByRole('heading', { name: '销售经理看板' })).toBeVisible();
    await expect(page.getByText('全店销售漏斗')).toBeVisible();
    await expect(page.getByText('异常卡点警告')).toBeVisible();
    
    // 验证不显示其他角色的内容
    await expect(page.getByText('销售代表看板')).not.toBeVisible();
    await expect(page.getByText('管理员看板')).not.toBeVisible();
  });
  
  test('Given 管理员登录, When 访问首页, Then 显示管理员看板', async ({ page }) => {
    // Given: 使用管理员账号登录
    await page.goto('/api/auth/login?username=admin-user');
    
    // When: 访问首页
    await page.goto('/');
    
    // Then: 验证显示管理员看板
    await expect(page.getByRole('heading', { name: '管理员看板' })).toBeVisible();
    await expect(page.getByText('知识库管理')).toBeVisible();
    await expect(page.getByText('知识库上传功能即将上线')).toBeVisible();
    
    // 验证不显示其他角色的内容
    await expect(page.getByText('销售代表看板')).not.toBeVisible();
    await expect(page.getByText('销售经理看板')).not.toBeVisible();
  });
});
```
